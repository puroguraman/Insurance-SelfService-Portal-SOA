-- Πίνακας Πελατών
CREATE TABLE Customer (
cust_id BIGINT AUTO_INCREMENT PRIMARY KEY,
username VARCHAR(50) NOT NULL UNIQUE,
full_name VARCHAR(100),
afm VARCHAR(20)
);


-- Πίνακας Συμβολαίων
CREATE TABLE Policy (
policy_id BIGINT AUTO_INCREMENT PRIMARY KEY,
cust_id BIGINT,
category VARCHAR(20),
base_premium DECIMAL(10, 2),
FOREIGN KEY (cust_id) REFERENCES Customer(cust_id)
);


-- Πίνακας Καλύψεων
CREATE TABLE Coverage (
cov_id BIGINT AUTO_INCREMENT PRIMARY KEY,
description VARCHAR(100),
cost DECIMAL(10, 2)
);


-- Πίνακας Πληρωμών
CREATE TABLE Payment (
pay_id BIGINT AUTO_INCREMENT PRIMARY KEY,
policy_id BIGINT,
amount DECIMAL(10, 2),
payment_date DATE,
status VARCHAR(20), -- 'COMPLETED', 'PENDING'
FOREIGN KEY (policy_id) REFERENCES Policy(policy_id)
);



-- Πίνακας Καταγραφής Ενεργειών (Audit Log)
CREATE TABLE AuditLog (
log_id BIGINT AUTO_INCREMENT PRIMARY KEY,
cust_id BIGINT, -- Ποιος έκανε την ενέργεια
action VARCHAR(100), -- Τι ενέργεια έγινε (π.χ.
'UPDATE_COVERAGE')
status VARCHAR(20), -- Αποτέλεσμα (π.χ. 'SUCCESS', 'FAILED')
ip_address VARCHAR(45), -- Η IP του χρήστη για λόγους ασφαλείας
timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
FOREIGN KEY (cust_id) REFERENCES Customer(cust_id)
);
